import asyncio
from typing import List, Dict, Any
from datetime import datetime
from src.database.models import TestCase, TestExecution, TestResult
from src.utils.logger import setup_logger

logger = setup_logger(__name__)

class TestEngine:
    """Motor principal de ejecución de pruebas"""
    def __init__(self):
        self.executions = []
        self.metrics = {}
    
    async def execute_test_suite(self, suite_id: str, config: Dict[str, Any]) -> Dict:
        """Ejecuta una suite de pruebas completa"""
        logger.info(f"Starting test suite execution: {suite_id}")
        
        execution = TestExecution(
            suite_id=suite_id,
            started_at=datetime.utcnow(),
            status="running",
            config=config
        )
        
        try:
            # 1. Preparar entorno
            await self._prepare_environment(config)
            
            # 2. Cargar casos de prueba
            test_cases = await self._load_test_cases(suite_id)
            
            # 3. Ejecutar en paralelo
            results = await self._execute_parallel(test_cases, config)
            
            # 4. Analizar resultados
            analysis = await self._analyze_results(results)
            
            # 5. Generar reportes
            report = await self._generate_report(execution, results, analysis)
            
            execution.status = "completed"
            execution.completed_at = datetime.utcnow()
            
            logger.info(f"Test suite completed: {suite_id}")
            return report
            
        except Exception as e:
            logger.error(f"Test suite failed: {e}")
            execution.status = "failed"
            execution.error = str(e)
            raise
    
    async def _execute_parallel(self, test_cases: List[TestCase], config: Dict) -> List[TestResult]:
        """Ejecuta pruebas en paralelo"""
        workers = config.get("parallel_workers", 4)
        semaphore = asyncio.Semaphore(workers)
        
        async def execute_with_semaphore(test_case):
            async with semaphore:
                return await self._execute_single_test(test_case)
        
        tasks = [execute_with_semaphore(tc) for tc in test_cases]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        return [r for r in results if not isinstance(r, Exception)]
    
    async def _analyze_results(self, results: List[TestResult]) -> Dict:
        """Analiza resultados de pruebas"""
        total = len(results)
        passed = sum(1 for r in results if r.status == "passed")
        failed = total - passed
        
        return {
            "total": total,
            "passed": passed,
            "failed": failed,
            "pass_rate": (passed / total * 100) if total > 0 else 0
        }

import asyncio
from typing import List, Dict, Any
from openai import AsyncOpenAI
from src.utils.logger import setup_logger
from src.ai.analyzers.requirement_analyzer import RequirementAnalyzer
from src.ai.analyzers.risk_analyzer import RiskAnalyzer

logger = setup_logger(__name__)

class AITestGenerator:
    """Generador de casos de prueba con IA"""
    def __init__(self, openai_api_key: str, model: str = "gpt-4"):
        self.client = AsyncOpenAI(api_key=openai_api_key)
        self.model = model
        self.requirement_analyzer = RequirementAnalyzer()
        self.risk_analyzer = RiskAnalyzer()
    
    async def generate_from_user_story(
        self,
        user_story: str,
        epic_context: str = None,
        platform: str = "mobile",
        test_type: str = "functional"
    ) -> List[Dict[str, Any]]:
        """Genera casos de prueba desde una User Story"""
        logger.info(f"Generating test cases from user story...")
        
        # 1. Analizar requisitos
        requirements = await self.requirement_analyzer.analyze(user_story)
        
        # 2. Evaluar riesgos
        risks = await self.risk_analyzer.assess_risks(requirements)
        
        # 3. Generar casos con GPT-4
        test_cases = await self._generate_with_gpt(
            requirements, risks, epic_context, platform, test_type
        )
        
        # 4. Validar y optimizar
        optimized_cases = await self._validate_and_optimize(test_cases)
        
        logger.info(f"Generated {len(optimized_cases)} test cases")
        return optimized_cases
    
    async def _generate_with_gpt(self, requirements: Dict, risks: Dict, 
                                  epic_context: str, platform: str, test_type: str) -> List[Dict]:
        """Genera casos usando GPT-4"""
        prompt = self._build_prompt(requirements, risks, epic_context, platform, test_type)
        
        response = await self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": self._get_system_prompt()},
                {"role": "user", "content": prompt}
            ],
            temperature=0.3,
            max_tokens=2048
        )
        
        return self._parse_response(response.choices[0].message.content)
    
    def _get_system_prompt(self) -> str:
        return """Eres un experto en QA y testing de aplicaciones móviles. 
        Tu tarea es generar casos de prueba exhaustivos, incluyendo:
        - Casos felices (happy path)
        - Casos negativos y de error
        - Casos de borde (edge cases)
        - Pruebas de seguridad
        - Pruebas de rendimiento
        
        Genera casos claros, ejecutables y con pasos detallados."""
    
    def _parse_response(self, response: str) -> List[Dict]:
        """Parsea la respuesta de GPT"""
        import json
        import re
        
        # Extraer JSON de la respuesta
        json_match = re.search(r'\[.*\]', response, re.DOTALL)
        if json_match:
            try:
                return json.loads(json_match.group())
            except json.JSONDecodeError:
                logger.error("Failed to parse GPT response as JSON")
                return []
        return []

import re
from typing import Dict, List
from src.utils.logger import setup_logger

logger = setup_logger(__name__)

class RequirementAnalyzer:
    """Analizador de requisitos con NLP"""
    
    async def analyze(self, user_story: str) -> Dict:
        """Analiza una User Story y extrae requisitos"""
        logger.info("Analyzing requirements...")
        
        return {
            "description": user_story,
            "acceptance_criteria": self._extract_acceptance_criteria(user_story),
            "functional_requirements": self._extract_functional_reqs(user_story),
            "non_functional_requirements": self._extract_non_functional_reqs(user_story),
            "entities": self._extract_entities(user_story),
            "actions": self._extract_actions(user_story)
        }
    
    def _extract_acceptance_criteria(self, text: str) -> List[str]:
        """Extrae criterios de aceptación"""
        patterns = [
            r"Criterios?\s*de\s*aceptaci[oó]n[:\s]*(.*?)(?:\n\n|$)",
            r"Acceptance\s*Criteria[:\s]*(.*?)(?:\n\n|$)",
            r"-\s*(.*?aceptaci[oó]n.*?)(?:\n|$)"
        ]
        
        criteria = []
        for pattern in patterns:
            matches = re.findall(pattern, text, re.IGNORECASE | re.DOTALL)
            criteria.extend(matches)
        
        return criteria if criteria else ["No especificados"]
    
    def _extract_functional_reqs(self, text: str) -> List[str]:
        """Extrae requisitos funcionales"""
        keywords = ["debe", "puede", "permite", "requiere", "necesita"]
        sentences = text.split(".")
        
        return [
            s.strip() for s in sentences
            if any(kw in s.lower() for kw in keywords)
        ]

class JiraClient:

        self.email = config["jira"]["email"]

        issue = self.client.issue(issue_key)
        return {
            "key": issue.key,
            "summary": issue.fields.summary,
            "description": issue.fields.description,
            "status": issue.fields.status.name,
            "priority": issue.fields.priority.name,
            "assignee": issue.fields.assignee.displayName if issue.fields.assignee else None,
            "labels": issue.fields.labels,
            "story_points": issue.fields.get("story_points", 0)
        }

        self,
        summary: str,
        description: str,
        test_steps: List[Dict],
        linked_story: str,

    ) -> Dict:

        issue_dict = {
            "project": {"key": self.project_key},
            "summary": summary,
            "description": description,
            "issuetype": {"name": "Test Case"},
            "labels": labels
        }

                "Executes",
                new_issue.key,

import re

from src.utils.logger import setup_logger

class DataClassifier:
    """Clasificador automático de datos sensibles"""
    
    def __init__(self):
        self.pii_patterns = self._load_pii_patterns()
    
    async def classify_data(self, data_sample: str) -> Dict:
        """Clasifica datos y detecta PII"""
        classification = {
            "sensitivity_level": "public",
            "pii_detected": False,
            "pii_types": [],
            "compliance_requirements": [],
            "recommended_actions": []
        }
        
        # 1. Detección de PII
        pii_matches = self.detect_pii_patterns(data_sample)
        if pii_matches:
            classification["pii_detected"] = True
            classification["pii_types"] = list(pii_matches.keys())
            classification["sensitivity_level"] = "sensitive"
            
            # 2. Clasificación de nivel
            if any(t in pii_matches for t in ["credit_card", "ssn"]):
                classification["sensitivity_level"] = "critical"
            
            # 3. Requisitos de compliance
            if classification["pii_detected"]:
                classification["compliance_requirements"] = ["GDPR", "CCPA"]
                classification["recommended_actions"] = [
                    "Enmascarar datos antes de usar en tests",
                    "Usar datos sintéticos en lugar de reales",
                    "Cifrar datos en reposo"
                ]
        
        return classification
    
    def detect_pii_patterns(self, data: str) -> Dict:
        """Detecta patrones de PII"""
        patterns = {
            "email": r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b',
            "phone": r'\+?1?\d{9,15}',
            "ssn": r'\d{3}-?\d{2}-?\d{4}',
            "credit_card": r'\d{4}[-\s]?\d{4}[-\s]?\d{4}[-\s]?\d{4}',
            "passport": r'[A-Z]{1,2}\d{6,9}'
        }
        
        detected = {}
        for pattern_type, pattern in patterns.items():
            matches = re.findall(pattern, str(data))
            if matches:
                detected[pattern_type] = matches
        
        return detected

from typing import Dict, Any
from src.utils.logger import setup_logger

logger = setup_logger(__name__)

class DataMaskingEngine:
    """Motor de enmascaramiento de datos"""
    
    async def mask_sensitive_data(self, data: Dict, masking_rules: Dict) -> Dict:
        """Aplica enmascaramiento según reglas"""
        masked_data = data.copy()
        
        for field_path, masking_config in masking_rules.items():
            field_value = self._get_nested_field(data, field_path)
            if field_value is not None:
                masked_value = await self._apply_masking_strategy(
                    field_value, masking_config
                )
                self._set_nested_field(masked_data, field_path, masked_value)
        
        return masked_data
    
    async def _apply_masking_strategy(self, value: Any, config: Dict) -> Any:
        """Aplica estrategia de enmascaramiento"""
        strategy = config["strategy"]
        
        strategies = {
            "full_mask": lambda v: "*" * len(str(v)),
            "partial_mask": self._partial_mask,
            "hash": self._hash_value,
            "encrypt": self._encrypt_value,
            "synthetic_replace": self._synthetic_replace,
            "format_preserve": self._format_preserve_mask
        }
        
        if strategy in strategies:
            return strategies[strategy](value, config.get("options", {}))
        else:
            raise ValueError(f"Unknown masking strategy: {strategy}")
    
    def _partial_mask(self, value: Any, options: Dict) -> str:
        """Enmascara parcialmente manteniendo algunos caracteres"""
        mask_char = options.get("mask_char", "*")
        preserve_start = options.get("preserve_start", 2)
        preserve_end = options.get("preserve_end", 2)
        
        str_value = str(value)
        if len(str_value) <= preserve_start + preserve_end:
            return mask_char * len(str_value)
        
        start = str_value[:preserve_start]
        end = str_value[-preserve_end:] if preserve_end > 0 else ""
        middle = mask_char * (len(str_value) - preserve_start - preserve_end)
        
        return start + middle + end
    
    def _synthetic_replace(self, value: Any, options: Dict) -> str:
        """Reemplaza con dato sintético"""
        from faker import Faker
        faker = Faker()
        
        field_type = options.get("field_type", "text")
        replacements = {
            "email": faker.email,
            "phone": faker.phone_number,
            "name": faker.name,
            "address": faker.address,
            "text": lambda: faker.text(max_nb_chars=50)
        }
        
        return replacements.get(field_type, lambda: faker.text())()

    """Runner de pruebas con Playwright"""
    
    def __init__(self, headless: bool = True):
        self.headless = headless
        self.browser = None
        self.context = None
        self.page = None
    
    async def execute(self, test_case: Dict, result: Dict) -> Dict:
        """Ejecuta un caso de prueba"""
        async with async_playwright() as p:
            try:
                # Lanzar navegador
                self.browser = await p.chromium.launch(headless=self.headless)
                self.context = await self.browser.new_context(
                    viewport={"width": 375, "height": 667},  # Mobile
                    user_agent="Mozilla/5.0 (iPhone; CPU iPhone OS 14_0 like Mac OS X)"
                )
                self.page = await self.context.new_page()
                
                # Ejecutar pasos
                for step in test_case.get("steps", []):
                    await self._execute_step(step)
                
                result["status"] = "passed"
                
            except Exception as e:
                result["status"] = "failed"
                result["error"] = str(e)
                
                # Capturar screenshot en caso de fallo
                if self.page:
                    await self.page.screenshot(path=f"failure_{test_case['id']}.png")
            
            finally:
                if self.browser:
                    await self.browser.close()
        
        return result
    
    async def _execute_step(self, step: Dict):
        """Ejecuta un paso del test"""
        action = step.get("action", "")
        target = step.get("target", "")
        value = step.get("value", "")
        
        if action == "navigate":
            await self.page.goto(target)
        elif action == "click":
            await self.page.click(target)
        elif action == "fill":
            await self.page.fill(target, value)
        elif action == "type":
            await self.page.type(target, value)
        elif action == "select":
            await self.page.select_option(target, value)
        elif action == "check":
            await self.page.check(target)
        elif action == "wait":
            await self.page.wait_for_timeout(int(value))
        elif action == "assert_visible":
            await self.page.wait_for_selector(target, state="visible")
        elif action == "assert_text":
            await self.page.wait_for_selector(f"text={value}")
        else:
            logger.warning(f"Unknown action: {action}")

from sqlalchemy import Column, Integer, String, DateTime, Text, JSON, Boolean, ForeignKey
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import relationship
from datetime import datetime

Base = declarative_base()

class TestCase(Base):
    __tablename__ = "test_cases"
    
    id = Column(Integer, primary_key=True, index=True)
    external_id = Column(String, unique=True, index=True)
    title = Column(String, nullable=False)
    description = Column(Text)
    type = Column(String, nullable=False)  # functional, api, performance, security
    priority = Column(String, nullable=False)  # critical, high, medium, low
    status = Column(String, default="pending")  # pending, passed, failed, blocked
    platform = Column(String, nullable=False)  # android, ios, mobile
    module = Column(String)
    steps = Column(JSON)
    test_data = Column(JSON)
    preconditions = Column(Text)
    postconditions = Column(Text)
    ai_generated = Column(Boolean, default=False)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    
    # Relationships
    executions = relationship("TestExecution", back_populates="test_case")

class TestExecution(Base):
    __tablename__ = "test_executions"
    
    id = Column(Integer, primary_key=True, index=True)
    suite_id = Column(String, nullable=False)
    test_case_id = Column(Integer, ForeignKey("test_cases.id"))
    status = Column(String, nullable=False)  # running, passed, failed, error
    started_at = Column(DateTime, default=datetime.utcnow)
    completed_at = Column(DateTime)
    duration_seconds = Column(Integer)
    error_message = Column(Text)
    screenshot_path = Column(String)
    logs = Column(Text)
    config = Column(JSON)
    
    # Relationships
    test_case = relationship("TestCase", back_populates="executions")
    results = relationship("TestResult", back_populates="execution")

class TestResult(Base):
    __tablename__ = "test_results"
    
    id = Column(Integer, primary_key=True, index=True)
    execution_id = Column(Integer, ForeignKey("test_executions.id"))
    step_number = Column(Integer)
    action = Column(String)
    expected = Column(Text)
    actual = Column(Text)
    status = Column(String, nullable=False)
    timestamp = Column(DateTime, default=datetime.utcnow)
    
    # Relationships
    execution = relationship("TestExecution", back_populates="results")

class JiraSync(Base):
    __tablename__ = "jira_sync"
    
    id = Column(Integer, primary_key=True, index=True)
    jira_issue_key = Column(String, unique=True, index=True)
    test_case_id = Column(Integer, ForeignKey("test_cases.id"))
    sync_status = Column(String, default="pending")
    last_synced_at = Column(DateTime)
    sync_data = Column(JSON)
